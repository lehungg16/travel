<script>
// LANGUAGE
let isVI = true;

function toggleLang(){
    isVI = !isVI;

    document.querySelectorAll("[data-vi]").forEach(el=>{
        el.innerText = isVI ? el.dataset.vi : el.dataset.en;
    });

    document.getElementById("heroTitle").innerText = isVI 
        ? "Khám phá Gia Lai xanh mát"
        : "Explore Green Gia Lai";

    document.getElementById("heroBtn").innerText = isVI 
        ? "Khám phá ngay"
        : "Explore now";
}

// SCROLL ANIMATION
let fades = document.querySelectorAll(".fade");

window.addEventListener("scroll", ()=>{
    fades.forEach(el=>{
        let top = el.getBoundingClientRect().top;
        if(top < window.innerHeight - 50){
            el.classList.add("show");
        }
    });
});
</script>
